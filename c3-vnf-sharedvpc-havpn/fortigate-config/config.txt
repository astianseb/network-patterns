config system interface
    edit "port1"
        set alias "ext"
end

config system interface
    edit "port2"
        set alias "int"
end
  
config firewall vip
    edit "sg-webserver-http"
        set extip 10.100.1.3
        set mappedip "10.10.1.2"
        set extintf "port1"
        set portforward enable
        set extport 80
        set mappedport 80
    next
end

config firewall policy
    edit 1
        set name "sg-port-forwarding"
        set srcintf "port1"
        set dstintf "port2"
        set action accept
        set srcaddr "all"
        set dstaddr "sg-webserver-http"
        set schedule "always"
        set service "HTTP"
        set nat enable
    next
end
